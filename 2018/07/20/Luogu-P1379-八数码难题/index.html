Luogu P1379 八数码难题

原题：Luogu 1379

蒟蒻第一次题解

这道题如果不想TLE和MLE的话，你需要判重。

所以有两种思路

因为这是一个九个点的矩阵，矩阵又涉及到0~8九个数字的可能

所以我们有了最最暴力的思路————

大力数组出奇迹

问题是，我们建一个到最大九位数排列可能性的数组，它除了包括你的九个数字还带上了一堆东西————1~最小可能性！

惊不惊喜，意不意外？没有，**

所以，我们为了避免这种可能，做出了比暴力稍微好一点的思路。

九维数组

什么你说你只用过三维数组？那你得记笔记了，你可以开一大堆维数组，只要不MLE。

九维数组代码如下

#include<bits/stdc++.h>
using namespace std;
bool bmap[9][9][9][9][9][9][9][9][9];
struct node {
    int state[3][3];
    int step;
    int x0,y0;
    int op(void) {
        int ans=0;
        for(int i=0; i<3; i++) {
            for(int i2=0; i2<3; i2++) {
                ans+=state[i][i2];
                ans*=10;
            }
        }
        ans/=10;
        return ans;
    }
    void csh(void) {
        for(int i=0; i<3; i++) {
            for(int i2=0; i2<3; i2++) {
                if(state[i][i2]==0) {
                    x0=i;
                    y0=i2;
                    return;
                }
            }
        }
    }
    void mov(int x,int y) {
        int a=state[x][y];
        int b=state[x0][y0];
        state[x][y]=b;
        state[x0][y0]=a;
        csh();
        return;
    }
    bool check(void){
        if(bmap[state[0][0]][state[0][1]][state[0][2]][state[1][0]][state[1][1]][state[1][3]][state[2][0]][state[2][1]][state[2][2]]==true)return true;
        else return false;
    }
    bool res(void){
        bmap[state[0][0]][state[0][1]][state[0][2]][state[1][0]][state[1][1]][state[1][3]][state[2][0]][state[2][1]][state[2][2]]=true;
    }
};
node start,father,son;
queue<node> q;
int main() {
    for(int i=0; i<3; i++) {
        for(int i2=0; i2<3; i2++) {
            start.state[i][i2]=getchar()-'0';
        }
    }
    start.csh();
    start.res();
    start.step=0;
    q.push(start);
    while(!q.empty()) {
        father=q.front();
        q.pop();
        /*
        for(int i=0; i<3; i++) {
            for(int i2=0; i2<3; i2++) {
                cout<<father.state[i][i2];
            }
            cout<<endl;
        }
        cout<<"x:"<<father.x0<<" y:"<<father.y0<<" step:"<<father.step<<endl;
        cout<<endl;
        //system("pause");*/
        if(father.op()==123804765) {
            cout<<father.step;
            return 0;
        }
        son=father;
        son.step++;
        if(father.x0!=0) {
            son.mov(father.x0-1,father.y0);
            if(son.check()!=true) {
                son.res();
                q.push(son);
            }
            son=father;son.step++;
        }
        if(father.x0!=2) {
            son.mov(father.x0+1,father.y0);
            if(son.check()!=true) {
                son.res();
                q.push(son);
            }
            son=father;son.step++;
        }
        if(father.y0!=0){
            son.mov(father.x0,father.y0-1);
            if(son.check()!=true) {
                son.res();
                    q.push(son);
            }
            son=father;son.step++;
        }
        if(father.y0!=2){
            son.mov(father.x0,father.y0+1);
            if(son.check()!=true) {
                son.res();
                q.push(son);
            }
            son=father;son.step++;
        }
    }
}
